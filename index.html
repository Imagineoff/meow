const express = require("express"); const bodyParser = require("body-parser"); const session = require("express-session"); const app = express(); app.use(bodyParser.urlencoded({ extended: true })); app.use(session({ secret: "tajnyKlic", resave: false, saveUninitialized: true })); // fake databáze const users = [{ username: "test", password: "heslo123" }]; const fakeResults = [ "Jak postavit vyhledávač od nuly", "Top 10 aut roku 2025", "Proč je Google tak rychlý" ]; // homepage app.get("/", (req, res) => { if (!req.session.user) { return res.send(`

Přihlášení

`); } res.send(`

Vítej ${req.session.user}

Odhlásit se `); }); // login app.post("/login", (req, res) => { const { username, password } = req.body; const user = users.find(u => u.username === username && u.password === password); if (user) { req.session.user = username; return res.redirect("/"); } res.send("Špatné jméno nebo heslo."); }); // search app.get("/search", (req, res) => { if (!req.session.user) return res.redirect("/"); const q = req.query.q || ""; const results = fakeResults.filter(r => r.toLowerCase().includes(q.toLowerCase())); res.send(`

Výsledky pro: "${q}"

    ${results.map(r => `
  • ${r}
    • `).join("")}
Zpět `); }); // logout app.get("/logout", (req, res) => { req.session.destroy(() => res.redirect("/")); }); app.listen(3000, () => console.log("Server běží na http://localhost:3000"));
